
// ICB ID Account Information
SELECT     U_MASTER.ID_AC AS ID_ACC_NO, DIVIDEND.BALANCE AS NO_OF_SHARE, DIVI_PARA.RATE, DIVIDEND.TOT_DIVI AS GROSS_DIVIDEND, 
                      DIVIDEND.DIDUCT AS TAX_DEDUCT, DIVIDEND.FI_DIVI_QTY AS NET_DIVIDEND, DIVI_PARA.F_YEAR AS FY, DIVI_PARA.CLOSE_DT, 
                      DIVI_PARA.CIP_RATE, DIVIDEND.CIP_QTY
FROM         DIVIDEND INNER JOIN
                      U_MASTER ON DIVIDEND.REG_BK = U_MASTER.REG_BK AND DIVIDEND.REG_BR = U_MASTER.REG_BR AND 
                      DIVIDEND.REG_NO = U_MASTER.REG_NO INNER JOIN
                      DIVI_PARA ON DIVIDEND.FUND_CD = DIVI_PARA.FUND_CD AND DIVIDEND.FY = DIVI_PARA.F_YEAR
WHERE     (DIVI_PARA.DIVI_NO = 8) AND (U_MASTER.ID_FLAG = 'Y') AND (U_MASTER.REG_BR = '01') AND (U_MASTER.CIP = 'N') AND 
                      (U_MASTER.HNAME LIKE 'ICB%')
