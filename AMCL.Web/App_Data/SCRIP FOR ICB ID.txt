
FOR NON CIP
CREATE TABLE ICB_ID_NCIP_7
AS
SELECT     U_MASTER.ID_AC AS ACCNT_NO, DIVIDEND.BALANCE AS NO_SHARE, DIVI_PARA.RATE, DIVIDEND.TOT_DIVI AS GROSS_AMOUNT, 
                      DIVIDEND.DIDUCT AS TAX_APP, DIVIDEND.FI_DIVI_QTY AS AMOUNT, DIVI_PARA.F_YEAR AS FYEAR, DIVI_PARA.CLOSE_DT
FROM         U_MASTER INNER JOIN
                      DIVIDEND ON U_MASTER.REG_BK = DIVIDEND.REG_BK AND U_MASTER.REG_BR = DIVIDEND.REG_BR AND 
                      U_MASTER.REG_NO = DIVIDEND.REG_NO INNER JOIN
                      DIVI_PARA ON DIVIDEND.FUND_CD = DIVI_PARA.FUND_CD AND DIVIDEND.FY = DIVI_PARA.F_YEAR AND 
                      DIVIDEND.DIVI_NO = DIVI_PARA.DIVI_NO
WHERE     (DIVI_PARA.DIVI_NO = 7) AND (U_MASTER.ID_FLAG = 'Y') AND (U_MASTER.REG_BR = '01') AND (U_MASTER.HNAME LIKE 'ICB%') AND 
                      (U_MASTER.CIP = 'N')



CREATE TABLE ICB_ID_CIP_7
AS
SELECT     U_MASTER.ID_AC AS ACCNT_NO, DIVIDEND.BALANCE AS NO_SHARE, DIVI_PARA.RATE, DIVIDEND.TOT_DIVI AS GROSS_AMOUNT, 
                      DIVIDEND.DIDUCT AS TAX_APP, DIVIDEND.FI_DIVI_QTY AS AMOUNT, DIVI_PARA.F_YEAR AS FYEAR, DIVI_PARA.CLOSE_DT, DIVI_PARA.CIP_RATE, 
                      DIVIDEND.CIP_QTY
FROM         U_MASTER INNER JOIN
                      DIVIDEND ON U_MASTER.REG_BK = DIVIDEND.REG_BK AND U_MASTER.REG_BR = DIVIDEND.REG_BR AND 
                      U_MASTER.REG_NO = DIVIDEND.REG_NO INNER JOIN
                      DIVI_PARA ON DIVIDEND.FUND_CD = DIVI_PARA.FUND_CD AND DIVIDEND.FY = DIVI_PARA.F_YEAR AND 
                      DIVIDEND.DIVI_NO = DIVI_PARA.DIVI_NO
WHERE     (DIVI_PARA.DIVI_NO = 7) AND (U_MASTER.ID_FLAG = 'Y') AND (U_MASTER.REG_BR = '01') AND (U_MASTER.HNAME LIKE 'ICB%') AND 
                      (U_MASTER.CIP = 'Y')
